  const [response] = await client.predict({
    endpoint,
    instances: [{ content: req.body.text }],
    parameters: { temperature: 0.3, maxOutputTokens: 150 }
  });
  const reply = response.predictions[0].generated_text || 'Lo siento, no entendÃ­.';
  res.json({ reply });
});
